{\Delta_1; \Gamma |- v_1 : A_1\\\\
\Delta_2; \Gamma |- v_2 : A_2}

%
\and
%
\Infer{inj}
{i \in \{1, 2\}\\\\

\and
%

{\Delta; \Gamma |- v : A}
{\Delta; \Gamma |- \Ref{v} : \tyM{A}}
%
\and
%
\Infer{thunk}
{\Delta; \Gamma |- e : A |> m}
{\Delta; \Gamma |- \thunk{e} : \tyUp{A, m}}
