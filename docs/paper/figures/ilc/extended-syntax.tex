  intuitionistic type $A$ and mode $m$.}

  \and
  %
  \Infer{tdp}
  {\Delta_1; \Gamma |- e_1 : [\tyBit]\\

  \judgbox{\Store_1 ; e_1 ---> \Store_2 ; e_2}{~~Under store $\Store_1$,
    expression~$e_1$ reduces to~$\Store_2 ; e_2$.}

  \Infer{kgen}
  {\keyword{\textbf{Gen}}(k, r) = (v_{pk}, v_{td}) \\ v_{k}, v_{td} \in \{0,1\}^n}

  \and
